鼠标显示一个层的同时隐藏其他层

代码如下：

<!DOCTYPE html>
<html>
<head>
<title></title>
<style type="text/css" media="screen">
    *{padding:0;margin: 0;}
    p{
        width:200px;
        height:200px;
        background: red;
    }
</style>
</head>
<body>
<a id="id1" style="display:none;" href="#" onMouseOver="show(0)">
    <div>1111</div>
</a>
<div id="child1" style="display:block;"><p></p></div>

<a id="id2" href="#" onMouseOver="show(1)">
    <div>2222</div>
</a>
<div id="child2" style="display:none;"><p></p></div>

<a id="id3" href="#" onMouseOver="show(2)">
    <div>3333</div>
</a>
<div id="child3" style="display:none;"><p></p></div>

<script language="javascript">
     function show(id){
        var aId= ["id1","id2","id3"];
        var div=["child1","child2","child3"];
        for(i=0;i<3;i++){
            document.getElementById(div[i]).style.display=(i==id?"block":"none");
            document.getElementById(aId[i]).style.display=(i==id?"none":"block");
        }
     }
</script>
</body>
</html>
